
Problem : A. Chewba—Åca and Number (Codeforces)

Problem link : https://codeforces.com/contest/514/problem/A

## Solution

```C++
    #include<iostream>
#include<string>
#include<map>
#include<vector>
#include<unordered_map>
using namespace std;

int main()
{
    string s;
    cin>>s;

    int n = s.size();

    int first = s[0] - '0';

    if(first != 9)
    {
        if(first > 4)
        {
            s[0] = ((9-first) + '0');
        }
    }

    for(int i=1;i<n;i++)
    {
        int d = s[i] - '0';

        if(d == 0 || d < 5)
           continue;
        else
           s[i] = ((9-d) + '0');
    }

    cout<<s<<endl;
    return 0;
}
```
